    disallowKeywordsOnNewLine

    Disallows placing keywords on a new line.

    Type: Array

    Values: Array of quoted keywords

    Example

    "disallowKeywordsOnNewLine": ["else"]
    Valid
    if (x < 0) {
    x++;
} else {
    x--;
}

    if (x < 0)
    x++;
else
  *     x--;

    if (x < 0) {
    x++;
}
// comments
else {
    x--;
}

    do {
    x++;
} while(x < 0);

    do
    x++;
while(x < 0);

    if (x < 0) {
    x++;

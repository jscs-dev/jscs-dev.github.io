    disallowNamedUnassignedFunctions

    Disallows unassigned functions to be named inline

    Type: Boolean

    Value: true

    Example

    "disallowNamedUnassignedFunctions": true
    Valid
    [].forEach(function () {});
var x = function() {};
function y() {}
    Invalid
    [].forEach(function x() {});

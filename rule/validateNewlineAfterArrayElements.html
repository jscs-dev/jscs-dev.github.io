    validateNewlineAfterArrayElements

    Requires each element in array on a single line when array length is more than passed maximum number or array fills more than one line. Set ignoreBrackets to true to allow elements on the same line with brackets.

    Type: Boolean or Number (maximum) or Object ({maximum: Number, ignoreBrackets: Boolean})

    Values: true

    Default: {maximum: Infinity, ignoreBrackets: false}

    Example

    "validateNewlineAfterArrayElements": {
  "maximum": 3
}
    Valid for true
    var x = [{a: 1}, [2], '3', 4, 5, 6];
var x = [
  {a: 1},
  [2],
  '3',
  4
];
    Invalid for true
    var x = [1,
  2];
    Valid for 3
    var x = [{a: 1}, [2], '3'];
var x = [
  1,
  2,
  3,
  4
];
    Invalid for 3
    var x = [1, 2, 3, 4];
var x = [1,
  2,
  3];
var x = [
    1, 2
];
    Valid for {maximum: 2, ignoreBrackets: true}
    var x = [{a: 1}, [2]];
var x = [1,
  2,
  3];
    Invalid for {maximum: 2, ignoreBrackets: true}
    var x = [1, 2, 3];
var x = [1, 2,
  3];
var x = [1,
  2, 3];

    disallowPaddingNewLinesAfterBlocks

    Disallow a newline after blocks

    Type: Boolean

    Value: true

    Example

    "disallowPaddingNewLinesAfterBlocks": true
    Valid
      * function () {
    for (var i = 0; i < 2; i++) {
        if (true) {
            return false;
        }
        continue;
    }
    var obj = {
        foo: function() {
            return 1;
        },
        bar: function() {
            return 2;
        }
    };
}
